    function lookupCorrection() {
        const x = parseInt(document.getElementById('inputX').value);
        const before = parseInt(document.getElementById('coinBefore').value);
        const after = parseInt(document.getElementById('coinAfter').value);
        const resultSpan = document.getElementById('resultValue');

        if (isNaN(x) || isNaN(before) || isNaN(after)) {
            resultSpan.innerText = '-';
            return;
        }

        // 獲得した純粋なコイン数
        const targetCoins = after - before;

        if (targetCoins < 0) {
            resultSpan.innerText = '入力エラー';
            return;
        }

        const rowData = rawTableData[x];
        if (!rowData) {
            resultSpan.innerText = '範囲外';
            return;
        }

        let bestIndex = null;
        let minDiff = Infinity;

        // 全ての列を検証して、最も数値が近いものを特定
        for (let i = 0; i < rowData.length; i++) {
            const diff = Math.abs(rowData[i] - targetCoins);
            if (diff < minDiff) {
                minDiff = diff;
                bestIndex = i;
            }
        }

        if (bestIndex !== null) {
            let matchedHeader = correctionHeaders[bestIndex];
            let displayText = '';

            if (matchedHeader === 'なし') {
                displayText = '補正 なし';
            } else {
                displayText = '補正 ' + matchedHeader;
            }
            
            // 例外処理：獲得コインが0枚で、複数の補正値（-10や-11など）が共に「0」で重複している場合
            if (targetCoins === 0 && rowData[bestIndex] === 0) {
                let matchingHeaders = [];
                for (let i = 0; i < rowData.length; i++) {
                    if (rowData[i] === 0) {
                        matchingHeaders.push(correctionHeaders[i]);
                    }
                }
                if (matchingHeaders.length > 1) {
                    displayText = '判定不可 (0枚重複)';
                }
            } else if (minDiff !== 0) {
                // 完全一致しない（少しブレがある）場合はアスタリスク
                displayText += ' *';
            }

            resultSpan.innerText = displayText;
        } else {
            resultSpan.innerText = '-';
        }
    }

    function init() {
        ['inputX', 'coinBefore', 'coinAfter'].forEach(id => {
            document.getElementById(id).addEventListener('input', lookupCorrection);
        });
    }

    window.onload = function() {
        init();
        lookupCorrection();
    };
